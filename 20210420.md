class Nation {
	String nation;
	String capital;
	int population;

	Nation(String nation, String capital, int population) {
		this.nation = nation;
		this.capital = capital;
		this.population = population;
	}

	@Override
	public String toString() {
		return "국가 : " + nation + ", 수도 : " + capital + ", 인구 : " + population;
	}
}
